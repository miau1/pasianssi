6   package pasianssi.pasianssi.gui;
7   
8   import java.awt.Dimension;
9   import javax.swing.Box;
10  import javax.swing.BoxLayout;
11  import javax.swing.JButton;
12  import javax.swing.JPanel;
13  
14  /**
15   * Yläpaneeli sisältää kaikki ylhäällä olevat pakat pasianssissa.
16   * 
17   * @author mikko
18   */
19  public class Ylapaneeli extends JPanel {
20  
21      private JPanel[] ylatp;
22      private JButton[] ylat;
23  
24      public Ylapaneeli(ImageHandler kuvat) {
25          this.setLayout(new BoxLayout(this, BoxLayout.X_AXIS));
26  
27          ylat = new JButton[6];
28  
29          this.add(Box.createRigidArea(new Dimension(20, 0)));
30  
31          for (int i = 0; i < 2; i++) {
32              ylat[i] = new JButton();
33              ylat[i].setPreferredSize(new Dimension(70, 95));
34              this.add(ylat[i]);
35              this.add(Box.createRigidArea(new Dimension(10, 0)));
36          }
37  
38          this.add(Box.createRigidArea(new Dimension(200, 0)));
39  
40          for (int i = 2; i < 6; i++) {
41              ylat[i] = new JButton();
42              ylat[i].setPreferredSize(new Dimension(70, 95));
43              this.add(ylat[i]);
44              this.add(Box.createRigidArea(new Dimension(10, 0)));
45          }
46  
47          ylat[0].setIcon(kuvat.kuva(0));
48          ylat[1].setIcon(kuvat.kuva(105));
49          for (int i = 2; i < 6; i++) {
50              ylat[i].setIcon(kuvat.kuva(105));
51          }
52      }
53  
54      public JButton[] getYlat() {
55          return ylat;
56      }
57  
58      public JPanel[] getYlatp() {
59          return ylatp;
60      }
61  
62  }
