6   package pasianssi.pasianssi.gui;

8   import javax.swing.JFrame;
9   import javax.swing.JLabel;
10  import javax.swing.SwingConstants;
11  import javax.swing.WindowConstants;
12  
13  /**
14   * Freimi, joka näyttää onnittelutekstin ja peliin kuluneen ajan, kun peli on
15   * voitettu.
16   * 
17   * @author mikko
18   */
19  public class VoittoFrame extends JFrame{
20      
21      private JLabel voittoLabel;
22      
23      public VoittoFrame(){
24          this.setTitle("VOITIT PELIN");
25          this.setSize(500, 250);
26          voittoLabel = new JLabel("ONNEKSI OLKOON!!1", SwingConstants.CENTER);
27          this.add(voittoLabel);
28          this.setDefaultCloseOperation(WindowConstants.HIDE_ON_CLOSE);
29          this.setVisible(false);
30      }
31      
32      public void vaihdaTeksti(String teksti){
33          voittoLabel.setText(teksti);
34      }
35      
36  }
