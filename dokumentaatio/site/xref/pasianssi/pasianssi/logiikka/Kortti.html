3   /**
4    * Luokka kuvaa yksittäistä pelikorttia, jolla on numero, maa ja puoli.
5    *
6    * @author mikko
7    */
8   public class Kortti {
9   
10      private int numero;
11      private int maa;
12      private int vari;
13      private int puoli;
14      private String kuva;
15  
16      /**
17       * Luo uuden kortin annetun numeron ja maan perusteella. Puoli on aluksi 0
18       * eli selkäpuoli ylöspäin. Antaa kortille värin maan perusteella.
19       *
20       * @param numero Annettu numero välillä 1-13.
21       * @param maa Annettu maa välillä 1-4.
22       */
23      public Kortti(int numero, int maa) {
24          this.numero = numero;
25          this.maa = maa;
26          vari = maa % 2;
27          this.puoli = 0;
28          this.kuva = kuvanOsoite(numero, maa);
29      }
30  
31      /**
32       * Metodi kääntää kortin toisinpäin vaihtamalla nollan ykköseksi tai ykkösen
33       * nollaksi.
34       */
35      public void kaanna() {
36          this.puoli = 1 - this.puoli;
37      }
38  
39      /**
40       * Palauttaa kortin numeron.
41       *
42       * @return Kortin numero.
43       */
44      public int getNumero() {
45          return numero;
46      }
47  
48      /**
49       * Palauttaa kortin maan. 1 = hertta, 2 = pata, 3 = ruutu ja 4 = risti.
50       *
51       * @return Kortin maa.
52       */
53      public int getMaa() {
54          return maa;
55      }
56  
57      /**
58       * Palauttaa kortin varin. 0 = musta ja 1 = punainen.
59       *
60       * @return Kortin vari.
61       */
62      public int getVari() {
63          return vari;
64      }
65  
66      /**
67       * Kertoo, kumpi puoli kortilla on ylöspäin. 0 = selkäpuoli ja 1 on
68       * kuvapuoli.
69       *
70       * @return Kortin puoli.
71       */
72      public int getPuoli() {
73          return puoli;
74      }
75  
76      /**
77       * Palauttaa kortin kuvan osoitteen.
78       *
79       * @return Kuvan osoite
80       */
81      public String getKuva() {
82          return kuva;
83      }
84  
85      /**
86       * Määrittää kuvan osoitteen maan ja numeron perusteella.
87       *
88       * @param num
89       * @param maa
90       * @return Kuvan osoite
91       */
92      private String kuvanOsoite(int num, int maa) {
93          String osoite = "src/main/resources/images/kortit/Cards/Classic/";
94          if (maa == 1) {
95              osoite = osoite + "h";
96          } else if (maa == 2) {
97              osoite = osoite + "s";
98          } else if (maa == 3) {
99              osoite = osoite + "d";
100         } else if (maa == 4) {
101             osoite = osoite + "c";
102         }
103         if (num < 10) {
104             osoite = osoite + "0";
105         }
106         osoite = osoite + Integer.toString(num) + ".png";
107         return osoite;
108     }
109 
110     public String toString() {
111         String maa2 = "";
112         if (this.maa == 1) {
113             maa2 = "Hertta";
114         } else if (this.maa == 2) {
115             maa2 = "Pata";
116         } else if (this.maa == 3) {
117             maa2 = "Ruutu";
118         } else if (this.maa == 4) {
119             maa2 = "Risti";
120         }
121         return maa2 + " " + Integer.toString(this.numero);
122     }
123 }
